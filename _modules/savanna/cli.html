[docs]def main(argv=None, config_files=None):
    CONF.register_cli_opt(command_opt)
    CONF(args=argv[1:],
         project='savanna',
         usage='%(prog)s [' + '|'.join(
             [cmd.name for cmd in CLI_COMMANDS]) + ']',
         default_config_files=config_files)
    log.setup("savanna")
    CONF.command.cmd_class.main()
